# 驗證事件要求{#validating-event-requests}

每個事件型別的JSON要求內文，都會在後端透過JSON結構描述進行驗證。 當API呼叫的驗證失敗時，HTTP回應內文會填入錯誤訊息。

每種事件類型的 JSON 驗證結構均可公開存取，請前往：`{uri}/api/v1/schemas/{eventType}` (例如 `{uri}/api/v1/schemas/sessionEnd`)。 JSON 驗證結構是判斷每種事件類型之正確要求內容參數的絕對可信來源。

例如，`sessionStart` 驗證結構之要求的回應將與以下範例相似 (稍微調整格式以利閱讀)：

```

Content-Type: application/json
Content-Length: 2716
Connection: keep-alive

{
  "$schema":"https://json-schema.org/draft-04/schema#",
  "id":"https://alpha.hb-api.omtrdc.net/api/v1/schemas/sessionStart",
  "definitions":
  {
    "playerTime":
    {
      "type":"object","properties":
      {
        "playhead":
        {"type":"number"},
        "ts":
        {"type":"integer"}
      },
      "required":["playhead","ts"],
      "additionalProperties":false
    },
    "eventType":
    {
      "type":"string",

        "sessionStart",
        "play",

        "pauseStart",
        "sessionComplete",

        "adBreakComplete",
        "adStart",

        "sessionEnd"
      ]
    },
    "qoeData":
    {
      "type":"object","properties":
      {
        "media.qoe.bitrate":
        {"type":"integer"},
        "media.qoe.droppedFrames":
        {"type":"integer"},
        "media.qoe.framesPerSecond":

        {"type":"integer"}
      },

      "additionalProperties":false
    },
    "customMetadata":

      {

      "properties":
      {

        {"type":"string","enum":["sessionStart"]},

        {"type":"object",

          {
            "appInstallationId":
            {"type":"string"},
            "analytics.trackingServer":
            {"type":"string"},

            "visitor.marketingCloudOrgId"],

      }

    "type":"object","$ref":"#/definitions/sessionStart"
  }
}
```
